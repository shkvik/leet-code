function twoSums(arr: number[], target: number){
  let left = 0, right = arr.length - 1;
  while(left < right){
    const curSum = arr[left] + arr[right];
    if(curSum === target){
      return [left, right];
    }
    else if(target > curSum){
      left++;
    }
    else{
      right--;
    }
  }
}
